# Show a word cloud of all keywords in tags

from wordcloud import STOPWORDS,WordCloud

table_videos.head()

tags = table_videos['tags'].map(lambda x:x.lower().split('|')).values

splt_tags = ' '.join(table_videos['tags'])

wordcloud = WordCloud(width=1000, height=500).generate(' '.join(splt_tags.lower().split('|')))

plt.figure(figsize=(15,5))
plt.imshow(wordcloud)
plt.axis('off')

plt.tight_layout()
plt.savefig('tags_word_cloud.png', dpi=300)
